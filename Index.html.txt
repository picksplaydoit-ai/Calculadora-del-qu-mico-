        function calculateRegression() {
            const xs = Array.from(document.querySelectorAll('.reg-x-input')).map(i => i.value).filter(v => v !== "").map(Number);
            const ys = Array.from(document.querySelectorAll('.reg-y-input')).map(i => i.value).filter(v => v !== "").map(Number);
            if(xs.length < 2 || xs.length !== ys.length) { regModel.hasData = false; return; }
            const n = xs.length; let sX = 0, sY = 0, sXY = 0, sXX = 0, sYY = 0;
            for(let i=0; i<n; i++) { sX += xs[i]; sY += ys[i]; sXY += xs[i]*ys[i]; sXX += xs[i]*xs[i]; sYY += ys[i]*ys[i]; }
            const den = (n * sXX - sX * sX); if(den === 0) return;
            const m = (n * sXY - sX * sY) / den; const b = (sY - m * sX) / n;
            const rN = (n * sXY - sX * sY); const rD = Math.sqrt((n * sXX - sX * sX) * (n * sYY - sY * sY));
            const r = rD !== 0 ? rN / rD : 0;
            regModel = { m, b, r, r2: r * r, hasData: true };
            updateRegressionUI(xs, ys);
        }
